SEED = 255

balanced_df = pd.DataFrame(data=None, columns=selected_df.columns)

for genre in selected_genres:
    data = selected_df.loc[selected_df['Genre'] == genre]
    balanced_df = balanced_df.append(data.sample(N, random_state=SEED))
